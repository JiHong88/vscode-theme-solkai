import { Request, Response } from 'express';

interface Session {
  token: string | null;
  verified: boolean;
  metadata?: Record<string, unknown>;
}

const TIMEOUT = 3000;

export function authenticate(req: Request): Session {
  const header = req.headers['authorization'];

  if (header === undefined) {
    return { token: null, verified: false };
  }

  const result = verifyJWT(header, TIMEOUT);
  const valid = result.payload !== null;

  return {
    token: result.payload,
    verified: true,
    metadata: { iat: Date.now() }
  };
}
